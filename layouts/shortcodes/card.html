{{- with .Err -}} {{- warnf "%s" . -}} {{/* Fallback to an anchor tag */}} {{ $url | safeHTML }} {{/* Otherwise, proceed with preview card generation. */}} {{- else -}} {{/* Scrape the page for meta {{- else -}} {{- partial "plugin-cards/cardify/preview-card.html" $params -}} {{- end -}} {{- end -}} {{/* with .Err */}} {{- end -}} {{/* with resources.GetRemote $url */}} {{- end }} {{/* if eq
